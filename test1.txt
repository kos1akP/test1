double pika(int k)
{
	int i = -3; int pr1; int pr; pr = 1; int n = i;
	do
	{
		if (n == 2)
		{
			cout << " не считали при n = 2" << endl;
		}
		else
		{
			pr1 = (pow(n, 3) - 8) / (n + 4);
			pr *= pr1;
			cout << "pr = " << pr << endl;
		}
		n++;
	} while (n <= 2 * k);
	return pr;
}
